PROCEDURE DIVISION USING LK-OPERATION END-OF-FILE EMPLOYEE.
DAL-READ.
    EXEC SQL
        DECLARE C_EMP CURSOR FOR
        SELECT EMP_ID, EMP_NAME, SALARY_BRUT, SALARY_NET
        FROM EMPLOYEE
    END-EXEC

    EXEC SQL
        OPEN C_EMP
    END-EXEC

    IF SQLCODE NOT = 0
        MOVE 'Y' TO END-OF-FILE
        GO TO DAL-READ-EXIT
    END-IF

    EXEC SQL
        FETCH C_EMP INTO
            :EMP-ID,
            :EMP-NAME,
            :SALARY-BRUT,
            :SALARY-NET
    END-EXEC

    IF SQLCODE = 0
        CONTINUE
    ELSE IF SQLCODE = 100
        MOVE 'Y' TO END-OF-FILE
    ELSE
        MOVE 'Y' TO END-OF-FILE
    END-IF

DAL-READ-EXIT.
    EXIT.

DAL-SAVE.
    EXEC SQL
        UPDATE EMPLOYEE
        SET SALARY_NET = :SALARY-NET
        WHERE EMP_ID = :EMP-ID
    END-EXEC
    .

DAL-END.
    EXEC SQL
        CLOSE C_EMP
    END-EXEC

    EXEC SQL
        COMMIT
    END-EXEC
    .
